## 方案一：AI自动生成封面（推荐）

### 1.1 使用Python脚本自动生成

创建 `tools/generate-cover.py`：

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
文章封面自动生成工具
支持多种风格：渐变、几何、抽象、科技等
"""

from PIL import Image, ImageDraw, ImageFont, ImageFilter
import random
import os
import argparse
import textwrap
from datetime import datetime

class CoverGenerator:
    def __init__(self, output_dir="source/img/covers"):
        self.output_dir = output_dir
        os.makedirs(output_dir, exist_ok=True)

        # 配色方案 - 苹果风格
        self.color_schemes = {
            "blue": [(0, 122, 255), (88, 86, 214)],
            "purple": [(175, 82, 222), (88, 86, 214)],
            "pink": [(255, 55, 95), (175, 82, 222)],
            "orange": [(255, 149, 0), (255, 59, 48)],
            "green": [(52, 199, 89), (48, 209, 88)],
            "teal": [(48, 176, 199), (90, 200, 250)],
            "dark": [(28, 28, 30), (44, 44, 46)],
            "light": [(242, 242, 247), (229, 229, 234)]
        }

        # 字体路径（需要下载相应字体）
        self.font_paths = {
            "title": "fonts/PingFang-SC-Bold.ttf",
            "subtitle": "fonts/PingFang-SC-Regular.ttf"
        }

    def generate_gradient_cover(self, title, subtitle="", style="blue", size=(1200, 630)):
        """
        生成渐变风格封面
        """
        # 创建画布
        img = Image.new('RGB', size, (255, 255, 255))
        draw = ImageDraw.Draw(img)

        # 获取配色
        colors = self.color_schemes.get(style, self.color_schemes["blue"])

        # 绘制渐变背景
        for y in range(size[1]):
            ratio = y / size[1]
            r = int(colors[0][0] * (1 - ratio) + colors[1][0] * ratio)
            g = int(colors[0][1] * (1 - ratio) + colors[1][1] * ratio)
            b = int(colors[0][2] * (1 - ratio) + colors[1][2] * ratio)
            draw.line([(0, y), (size[0], y)], fill=(r, g, b))

        # 添加装饰元素
        self._add_decorations(draw, size, colors[0])

        # 添加文字
        self._add_text(img, title, subtitle, size)

        # 添加毛玻璃效果区域
        self._add_glass_effect(img, size)

        return img

    def generate_geometric_cover(self, title, subtitle="", style="blue", size=(1200, 630)):
        """
        生成几何风格封面
        """
        img = Image.new('RGB', size, (255, 255, 255))
        draw = ImageDraw.Draw(img)

        colors = self.color_schemes.get(style, self.color_schemes["blue"])
        bg_color = (245, 245, 247)
        img.paste(bg_color, (0, 0, size[0], size[1]))

        # 绘制几何图形
        for _ in range(15):
            shape_type = random.choice(['circle', 'rect', 'triangle'])
            x = random.randint(0, size[0])
            y = random.randint(0, size[1])
            shape_size = random.randint(50, 200)
            color = colors[random.randint(0, 1)]
            alpha = random.randint(30, 80)

            if shape_type == 'circle':
                draw.ellipse([x, y, x+shape_size, y+shape_size], 
                           fill=(*color, alpha))
            elif shape_type == 'rect':
                draw.rectangle([x, y, x+shape_size, y+shape_size], 
                             fill=(*color, alpha))

        # 添加文字
        self._add_text(img, title, subtitle, size)

        return img

    def generate_abstract_cover(self, title, subtitle="", style="blue", size=(1200, 630)):
        """
        生成抽象艺术风格封面
        """
        img = Image.new('RGB', size, (255, 255, 255))
        draw = ImageDraw.Draw(img)

        colors = self.color_schemes.get(style, self.color_schemes["blue"])

        # 绘制流动的线条
        for i in range(20):
            points = []
            x = random.randint(0, size[0])
            y = random.randint(0, size[1])

            for _ in range(5):
                x += random.randint(-200, 200)
                y += random.randint(-100, 100)
                points.append((x, y))

            color = colors[i % 2]
            draw.line(points, fill=(*color, 100), width=random.randint(2, 8))

        # 添加模糊效果
        img = img.filter(ImageFilter.GaussianBlur(radius=2))

        # 添加文字
        self._add_text(img, title, subtitle, size)

        return img

    def generate_tech_cover(self, title, subtitle="", style="blue", size=(1200, 630)):
        """
        生成科技风格封面
        """
        img = Image.new('RGB', size, (10, 10, 20))
        draw = ImageDraw.Draw(img)

        colors = self.color_schemes.get(style, self.color_schemes["blue"])

        # 绘制网格
        grid_size = 40
        for x in range(0, size[0], grid_size):
            draw.line([(x, 0), (x, size[1])], fill=(30, 30, 50), width=1)
        for y in range(0, size[1], grid_size):
            draw.line([(0, y), (size[0], y)], fill=(30, 30, 50), width=1)

        # 绘制发光点
        for _ in range(50):
            x = random.randint(0, size[0])
            y = random.randint(0, size[1])
            color = colors[random.randint(0, 1)]
            draw.ellipse([x-2, y-2, x+2, y+2], fill=color)

        # 添加文字
        self._add_text(img, title, subtitle, size, text_color=(255, 255, 255))

        return img

    def _add_decorations(self, draw, size, color):
        """添加装饰元素"""
        # 添加圆形装饰
        for _ in range(5):
            x = random.randint(-100, size[0])
            y = random.randint(-100, size[1])
            radius = random.randint(100, 300)
            draw.ellipse([x, y, x+radius, y+radius], 
                        fill=(*color, 20))

    def _add_text(self, img, title, subtitle, size, text_color=None):
        """添加文字"""
        draw = ImageDraw.Draw(img)

        # 默认文字颜色
        if text_color is None:
            text_color = (255, 255, 255)

        # 加载字体（使用系统字体作为fallback）
        try:
            title_font = ImageFont.truetype(self.font_paths["title"], 72)
            subtitle_font = ImageFont.truetype(self.font_paths["subtitle"], 36)
        except:
            title_font = ImageFont.load_default()
            subtitle_font = ImageFont.load_default()

        # 处理标题换行
        max_width = size[0] - 200
        lines = textwrap.wrap(title, width=15)

        # 绘制标题
        y_offset = size[1] // 2 - (len(lines) * 80) // 2
        for line in lines:
            bbox = draw.textbbox((0, 0), line, font=title_font)
            text_width = bbox[2] - bbox[0]
            x = (size[0] - text_width) // 2
            draw.text((x, y_offset), line, font=title_font, fill=text_color)
            y_offset += 90

        # 绘制副标题
        if subtitle:
            y_offset += 20
            bbox = draw.textbbox((0, 0), subtitle, font=subtitle_font)
            text_width = bbox[2] - bbox[0]
            x = (size[0] - text_width) // 2
            draw.text((x, y_offset), subtitle, font=subtitle_font, 
                     fill=(*text_color, 200))

    def _add_glass_effect(self, img, size):
        """添加毛玻璃效果区域"""
        # 创建遮罩
        overlay = Image.new('RGBA', size, (255, 255, 255, 0))
        draw = ImageDraw.Draw(overlay)

        # 绘制半透明区域
        glass_height = 200
        y_start = size[1] // 2 - glass_height // 2
        draw.rectangle([100, y_start, size[0]-100, y_start+glass_height], 
                      fill=(255, 255, 255, 30))

        # 合并
        img = Image.alpha_composite(img.convert('RGBA'), overlay)
        return img.convert('RGB')

    def save_cover(self, img, filename):
        """保存封面"""
        filepath = os.path.join(self.output_dir, filename)
        img.save(filepath, 'JPEG', quality=95)
        print(f"✅ 封面已保存: {filepath}")
        return filepath

    def generate_for_post(self, title, subtitle="", style="blue", template="gradient"):
        """为文章生成封面"""
        # 生成安全文件名
        safe_title = "".join(c for c in title if c.isalnum() or c in (' ', '-')).strip()
        safe_title = safe_title.replace(' ', '-')
        filename = f"cover-{safe_title}-{datetime.now().strftime('%Y%m%d')}.jpg"

        # 选择模板
        generators = {
            "gradient": self.generate_gradient_cover,
            "geometric": self.generate_geometric_cover,
            "abstract": self.generate_abstract_cover,
            "tech": self.generate_tech_cover
        }

        generator = generators.get(template, self.generate_gradient_cover)
        img = generator(title, subtitle, style)

        return self.save_cover(img, filename)

def main():
    parser = argparse.ArgumentParser(description="文章封面生成工具")
    parser.add_argument("-t", "--title", required=True, help="文章标题")
    parser.add_argument("-s", "--subtitle", default="", help="副标题")
    parser.add_argument("--style", default="blue", 
                       choices=["blue", "purple", "pink", "orange", "green", "teal", "dark", "light"],
                       help="配色风格")
    parser.add_argument("--template", default="gradient",
                       choices=["gradient", "geometric", "abstract", "tech"],
                       help="封面模板")
    parser.add_argument("-o", "--output", default="source/img/covers", help="输出目录")

    args = parser.parse_args()

    generator = CoverGenerator(args.output)
    generator.generate_for_post(args.title, args.subtitle, args.style, args.template)

if __name__ == "__main__":
    main()
```

## 方案三：一键生成脚本（集成到发布流程）

### 3.1 增强版发布脚本

修改 `publish.py`，添加自动封面生成功能：

```python
def create_post_with_cover(self, title, categories=None, tags=None, 
                          post_type="article", description=None, 
                          top=False, cover_style="blue"):
    """
    创建文章并自动生成封面
    """
    # 生成封面
    from generate_cover import CoverGenerator

    cover_generator = CoverGenerator()
    subtitle = description or f"{datetime.datetime.now().strftime('%Y-%m-%d')}"

    # 根据文章类型选择风格
    style_map = {
        "tech": "blue",
        "life": "pink",
        "tutorial": "green",
        "article": "purple"
    }
    style = style_map.get(post_type, cover_style)

    cover_path = cover_generator.generate_for_post(title, subtitle, style)

    # 获取相对路径
    cover_url = "/" + cover_path.replace("source/", "")

    # 创建文章
    return self.create_post(
        title=title,
        categories=categories,
        tags=tags,
        post_type=post_type,
        cover=cover_url,
        description=description,
        top=top
    )
```

```bash

# 指定封面风格
python publish.py -t "生活感悟" --type life --cover-style pink
```

## 方案四：AI图像生成（Midjourney/DALL-E）

### 4.1 Midjourney提示词模板

```
Blog cover image, [主题], minimalist design, gradient background, 
clean typography, modern style, professional look, 
1200x630 aspect ratio, high quality, 4k --ar 1200:630
```

**示例**:
- 技术文章: `Blog cover image, programming code on screen, blue gradient background, minimalist design, modern tech style --ar 1200:630`
- 生活文章: `Blog cover image, coffee and book on wooden table, warm lighting, cozy atmosphere, soft colors --ar 1200:630`

### 4.2 自动化脚本

```python
import openai
import requests

def generate_cover_with_ai(title, api_key):
    """使用DALL-E生成封面"""
    openai.api_key = api_key

    prompt = f"Blog cover image for article titled '{title}', \
              minimalist design, professional look, clean and modern, \
              suitable for tech blog, 1200x630"

    response = openai.Image.create(
        prompt=prompt,
        n=1,
        size="1024x1024"
    )

    image_url = response['data'][0]['url']

    # 下载并调整尺寸
    img_data = requests.get(image_url).content
    with open('temp_cover.png', 'wb') as f:
        f.write(img_data)

    # 调整尺寸为1200x630
    img = Image.open('temp_cover.png')
    img = img.resize((1200, 630), Image.Resampling.LANCZOS)

    return img
```

```python
import requests
from PIL import Image, ImageDraw, ImageFont
import random

def generate_cover_from_unsplash(title, keyword="technology"):
    """从Unsplash获取图片并添加文字"""

    # Unsplash API（需要申请API Key）
    access_key = "你的Unsplash API Key"
    url = f"https://api.unsplash.com/photos/random?query={keyword}&client_id={access_key}"

    response = requests.get(url)
    data = response.json()

    # 下载图片
    image_url = data['urls']['regular']
    img_data = requests.get(image_url).content

    with open('temp_bg.jpg', 'wb') as f:
        f.write(img_data)

    # 打开并调整尺寸
    img = Image.open('temp_bg.jpg')
    img = img.resize((1200, 630), Image.Resampling.LANCZOS)

    # 添加暗色遮罩
    overlay = Image.new('RGBA', (1200, 630), (0, 0, 0, 100))
    img = Image.alpha_composite(img.convert('RGBA'), overlay)

    # 添加文字
    draw = ImageDraw.Draw(img)

    # 标题
    try:
        font = ImageFont.truetype("fonts/PingFang-SC-Bold.ttf", 60)
    except:
        font = ImageFont.load_default()

    # 文字居中
    bbox = draw.textbbox((0, 0), title, font=font)
    text_width = bbox[2] - bbox[0]
    x = (1200 - text_width) // 2
    y = 315 - 30

    draw.text((x, y), title, font=font, fill=(255, 255, 255))

    return img.convert('RGB')
```

## 封面设计规范

### 尺寸标准
- **推荐尺寸**: 1200 x 630 像素
- **比例**: 1.91:1（适合社交媒体分享）
- **最小尺寸**: 600 x 315 像素

### 设计原则
1. **简洁明了**: 避免过多元素
2. **文字清晰**: 标题易读
3. **配色协调**: 符合博客整体风格
4. **品牌一致**: 保持视觉统一
5. **移动端适配**: 确保小屏幕可读

### 配色建议
- 主色调: 与博客主题一致
- 对比度: 确保文字可读
- 深色模式: 准备两套配色
